function generateInvoice(orderId){ let o=orders.find(x=>x.id===orderId); if(!o){showToast('⚠️ Order not found');return;} let party=getPartyById(o.partyId); let content=document.getElementById('billContent'); let paid=getOrderTotalPaid(o); let balance=o.orderTotal-paid; let rows=(o.doors||[]).map((d,i)=>{ let sqft = calcSqft(d.w, d.h) * (d.qty||1); // کل sq.ft let amount = (parseFloat(d.doorAmount)||0); return ` ${i+1} ${esc(d.size||'')} ${d.qty||1} ${sqft.toFixed(1)} ${fmtRs(d.rate||0)} ${fmtRs(amount)} `; }).join(''); content.innerHTML=`

${businessProfile.name}

${businessProfile.tagline||''} · ${businessProfile.address||''} · ${businessProfile.phone||''}

Invoice: ${o.orderNo}
Date: ${o.date}
Party: ${party?.name||'Unknown'}
${party?.phone||''}
${rows}
# Size Qty Sq.ft Rate Amount
TOTAL ${fmtRs(o.orderTotal)}
Total: ${fmtRs(o.orderTotal)}
Paid: ${fmtRs(paid)}
Balance: ${fmtRs(balance)}
${businessProfile.footerNote||'Thank You for Your Business 🙏'}
`; openModal('billModal'); }
